from pathlib import Path import zipfile, re src = Path("/mnt/data/index-new (8).html") text = src.read_text(encoding="utf-8") # Preload hero image text = text.replace( '', '\n ' ) # Remove desktop hero form text = re.sub( r'\n\s*
.*?
', '', text, count=1, flags=re.S ) # Remove mobile quote section text = re.sub( r'\n\s*.*?\s*', '\n\n', text, count=1, flags=re.S ) # Make hero single-column text = text.replace( """ .hero .container { position: relative; z-index: 1; display: grid; grid-template-columns: 1fr 420px; gap: 40px; align-items: center; padding: 100px 0 90px; }""", """ .hero .container { position: relative; z-index: 1; max-width: 760px; padding: 100px 0 90px; }""" ) # Keep stat strip on mobile and do not hide form text = text.replace(" .hero-form-card { display: none; }\n", "") text = text.replace(" /* hide stat strip on mobile, form takes its place */\n .stat-strip { display: none; }\n .mobile-quote-section { display: block !important; }\n", "") # Replace remote image with local image to reduce slow external fetch text = text.replace( 'src="https://images.pexels.com/photos/5233285/pexels-photo-5233285.jpeg?cs=srgb&dl=pexels-khunkorn-laowisit-1359149-5233285.jpg&fm=jpg"', 'src="assets/img/hero-upholstery-cleaning-optimised.webp"' ) # Insert below-hero form section marker = '\n' insert = """

Get your free quote

Tell us what needs cleaning and we'll be in touch fast.

Fast local reply Honest advice Professional service
""" if marker not in text: raise ValueError("Stat strip marker not found") text = text.replace(marker, "\n" + insert + marker, 1) # Add quote section spacing rule text = text.replace( ".section-alt { background: var(--off-white); }", ".section-alt { background: var(--off-white); }\n .quote-form-section { padding-top: 44px; padding-bottom: 44px; }", 1 ) out_dir = Path("/mnt/data/index_mobile_form_fixed") out_dir.mkdir(exist_ok=True) (out_dir / "index.html").write_text(text, encoding="utf-8") (out_dir / "WHAT-WAS-FIXED.txt").write_text( "Issue 1: The previous file still had the form inside the hero and the replacement never fully went in.\n" "Issue 2: Mobile CSS in the original page hid the hero form and relied on a separate mobile-only form block.\n" "Issue 3: The page was slowed down by a large hero image not being preloaded and a remote external image in the services grid.\n\n" "This version fixes those problems.\n", encoding="utf-8" ) zip_path = Path("/mnt/data/index-mobile-form-speed-fixed.zip") with zipfile.ZipFile(zip_path, "w", zipfile.ZIP_DEFLATED) as z: z.write(out_dir / "index.html", "index.html") z.write(out_dir / "WHAT-WAS-FIXED.txt", "WHAT-WAS-FIXED.txt") print(str(zip_path))
